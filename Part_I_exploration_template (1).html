## Declaring the function to plot the count plots of catagorical variables, to see their distributions.
def plotCountplot(x, figsize=(12,8)):
    #plot size
    plt.figure(figsize=figsize)
    #plot color
    base_color = sb.color_palette()[0]
    #Countplot
    sb.countplot(data = loan, x = x,color = base_color);
    #xticks rotation
    plt.xticks(rotation=90)
    #xlabel
    plt.xlabel(x, weight = "bold", fontsize = 12)
    #ylabel
    plt.ylabel('Count',weight = "bold", fontsize = 12)
    #title
    plt.title('Distribution of '+x, weight = "bold", fontsize = 16)
    plt.show()
